iniciar(gato); //RELLENA EL ARREGLO CON CEROS

public static void iniciar(int gato[][]){ // METODO PARA INICIALIZAR EL ARREGLO EN CEROS

for (int i = 0; i < 5; i++) { //CICLO FOR PARA RECORRER EL ARREGLO
for (int j = 0; j < 5; j++) { //CICLO FOR PARA RECORRER EL ARREGLO
gato[i][j]= 0; //COLOCA UN "0" EN LA POSICION ACTUAL DEL ARREGLO
}
}

}
public static void tiroJugador(int gato[][]){ //METODO PARA QUE EL JUGADOR 1 TIRAR

Scanner sc= new Scanner(System.in); //DECLARACION DE VARIABLE DE TIPO SCANNER PARA LEER POR TECLADO
int x, y; // DECLARACION DE VARIABLES PARA PODER COLOCAR LAS POSICIONES EN EL ARREGLO
System.out.println("Jugador 1, ¿En qué posición quiere tirar?"); //IMPRIME LA PREGUNTA PARA SABER DONDE QUIERE TIRAR EL JUGADOR
for (int i = 0; i < 5; i++) { //CICLO FOR PARA RECORRER EL ARREGLO
for (int j = 0; j < 5; j++) { //CICLO FOR PARA RECORRER EL ARREGLO
System.out.println("Posición: ");
x=sc.nextInt(); // TOMA LA POSICION X
y=sc.nextInt(); //TOMA LA POSICION Y

if (gato[x][y]==0) { // IF PARA REVISAR SI LA POSICION ESTA DESOCUPADA
gato[x][y]= 1; // SI ESTA DESOCUPADA COLOCA UN "1"
}else{ // SI NO ESTA DESOCUPADA SE MUESTRA UN MENSAJE DE ERROR
System.out.println("Error, vuelva a tirar!"); // IMPRIME EL MENSAJE DE ERROR
tiroJugador(gato); //LLAMA AL METODO PARA QUE TIRE EL JUGADOR
imprimir(gato); // LLAMA AL METODO PARA IMPRIMIR EL TABLERO
ganaste(gato);

}
tiroMaquina(gato); //LLAMA AL METODO PARA QUE TIRE LA MAQUINA
imprimir(gato); //LLAMA AL METODO PARA IMPRIMIR EL TABLERO
ganaste(gato);
}
}
}
public static void tiroMaquina(int gato[][]){
for (int i = 0; i < 5; i++) { //CICLO FOR PARA RECORRER EL ARREGLO
for (int j = 0; j < 5; j++) { //CICLO FOR PARA RECORRER EL ARREGLO
if (gato[i][j]==0) { // SI LA POSICION ACTUAL DEL ARREGLO ES IGUAL A "0" SE COLOCA UN 2
gato[i][j]=2;
return; // RETURN PARA SACARNOS DE LOS CICLOS FOR
}
}
}
}
public static void imprimir(int gato[][]){ // METODO PARA IMPRIMIR EL TABLERO
for (int i = 0; i < 5; i++) { //CICLO FOR PARA RECORRER EL ARREGLO
for (int j = 0; j < 5; j++) { //CICLO FOR PARA RECORRER EL ARREGLO
System.out.print(gato[i][j]+" "); //IMPRIME LA POSICION ACTUAL DEL ARREGLO
}
System.out.println(""); //DA UN SALTO DE LINEA
}
}
